def logger(func):  
    """–î–µ–∫–æ—Ä–∞—Ç–æ—Ä, –≤—ã–≤–æ–¥—è—â–∏–π –∏–º—è —Ñ—É–Ω–∫—Ü–∏–∏."""  
    def wrapper(*args):  
        print(f"–í—ã–∑–æ–≤ —Ñ—É–Ω–∫—Ü–∏–∏: {func.__name__}")  
        return func(*args)  
    return wrapper  

@logger  
def add(a, b):  
    return a + b
